136  top_length = cv::norm(top_points.front() - top_points.back());
137  bottom_length = cv::norm(bottom_points.front() - bottom_points.back());
138 
139  emit gotTilts(horizontal_tilt, vertical_tilt);
140 
141  if (left_length > right_length)
142  horizontal_tilt = left_length / right_length - 1;
143  else
144  horizontal_tilt = -right_length / left_length + 1;
145 
146  if (top_length > bottom_length)
147  vertical_tilt = top_length / bottom_length - 1;
148  else
