87 void AbstractStereoCamera::reproject3D() {
88 
89  cv::Mat disparity_downscale;
90 
91  matcher->getDisparity(disparity_downscale);
92 
93  cv::abs(disparity_downscale);
94 
95  // By default the disparity maps are scaled by a factor of 16.0
96  disparity_downscale /= 16.0;
97 
98  if (Q.empty() || disparity_downscale.empty()) {
99  return;
100  }
101 
102  cv::reprojectImageTo3D(disparity_downscale, stereo_reprojected, Q, true);
103 
104  pcl::PointCloud<pcl::PointXYZRGB>::Ptr ptCloudTemp(
105  new pcl::PointCloud<pcl::PointXYZRGB>);
106 
107  pcl::PointXYZRGB point;
108  uint32_t rgb = 0;
109  uchar col = 0;
110 
111  point.x = 0;
112  point.y = 0;
113  point.z = 0;
114 
115  rgb = ((int)255) << 16 | ((int)255) << 8 | ((int)255);
116  point.rgb = *reinterpret_cast<float *>(&rgb);
117  ptCloudTemp->points.push_back(point);
118 
119  for (int i = 0; i < stereo_reprojected.rows; i++) {
120  float *reconst_ptr = stereo_reprojected.ptr<float>(i);
121  float *disp_ptr = matcher->disparity_lr.ptr<float>(i);
122  uchar *rgb_ptr = left_remapped.ptr<uchar>(i);
123 
124  // TODO: Figure out why this can trigger, assume the disparity map isn't generated
125  // when this function gets called.
126  if(!disp_ptr || !rgb_ptr || !reconst_ptr) return;
127 
128  for (int j = 0; j < stereo_reprojected.cols; j++) {
129  if (disp_ptr[j] == 0) continue;
130  if (rgb_ptr[j] == 0) continue;
131 
132  point.x = reconst_ptr[3 * j];
133  point.y = -reconst_ptr[3 * j + 1];
134  point.z = -reconst_ptr[3 * j + 2];
135 
136  if(abs(point.x) > 10) continue;
137  if(abs(point.y) > 10) continue;
138  if(point.z == -10000) continue;
139  col = rgb_ptr[j];
140 
141  rgb = ((int)col) << 16 | ((int)col) << 8 | ((int)col);
142  point.rgb = *reinterpret_cast<float *>(&rgb);
143 
144  ptCloudTemp->points.push_back(point);
145  }
146  }
147 
148  if(ptCloudTemp->points.empty())
149  return;
150 
151  pcl::PassThrough<pcl::PointXYZRGB> pass;
152  pass.setInputCloud (ptCloudTemp);
153  pass.setFilterFieldName ("z");
154  pass.setFilterLimits(-visualisation_max_z, -visualisation_min_z);
155  pass.filter(*ptCloudTemp);
156 
157  /*
158  pcl::RadiusOutlierRemoval<pcl::PointXYZRGB> outrem;
159  outrem.setInputCloud(ptCloudTemp);
160  outrem.setRadiusSearch(0.05);
161  outrem.setMinNeighborsInRadius (5);
162  outrem.filter (*ptCloudTemp);
163 
164  pcl::StatisticalOutlierRemoval<pcl::PointXYZRGB> sor;
165  sor.setInputCloud (ptCloudTemp);
166  sor.setMeanK (5);
167  sor.setStddevMulThresh (4.0);
168  sor.filter(*ptCloudTemp);
169  */
170 
171  ptCloud = ptCloudTemp;
172 
173  emit reprojected();
174 }

256 bool AbstractStereoCamera::loadCalibration(QString directory) {
257  if (directory == "") return false;
258 
259  directory = QDir::cleanPath(directory);
260 
261  if (!loadRectificationMaps(directory + "/left_rectification.xml",
262  directory + "/right_rectification.xml")) {
263  rectification_valid = false;
264  qDebug() << "Couldn't find rectification maps";
265  return false;

267 
268  if (rectmapx_l.cols != image_width || rectmapx_r.cols != image_width ||
269  rectmapy_l.rows != image_height || rectmapy_r.rows != image_height) {
270  rectification_valid = false;
271  qDebug() << "Image size doesn't match rectification maps";
272 
273  return false;
274  }
275 
276  if (!loadCalibration(directory + "/left_calibration.xml",
277  directory + "/right_calibration.xml",
278  directory + "/stereo_calibration.xml")) {
279  qDebug() << "Couldn't load camera calibration";
280  calibration_valid = false;
281  return false;
282  }
283 
284  calibration_valid = true;
285  rectification_valid = true;
286 
287  enableRectify(true);
288 
289  return true;
