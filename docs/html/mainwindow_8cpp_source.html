247 void MainWindow::stereoCameraLoad(void) {
248  cameras_connected = false;
249 
250  StereoCameraPhobos* stereo_cam_phobos = new StereoCameraPhobos;
251  QThread* phobos_thread = new QThread;
252  stereo_cam_phobos->assignThread(phobos_thread);
253 
254  StereoCameraDeimos* stereo_cam_deimos = new StereoCameraDeimos;
255  QThread* deimos_thread = new QThread;
256  stereo_cam_deimos->assignThread(deimos_thread);
257 
258  if(stereo_cam_phobos->find_systems()){
259  stereo_cam = (AbstractStereoCamera*) stereo_cam_phobos;
260  left_view->setSettingsCallback(stereo_cam, SLOT(loadLeftSettings()));
261  right_view->setSettingsCallback(stereo_cam, SLOT(loadRightSettings()));
262  qDebug() << "Connecting to Phobos system";
263  cameras_connected = true;
264  }else if (stereo_cam_deimos->find_systems()) {
265  stereo_cam = (AbstractStereoCamera*) stereo_cam_deimos;
266  qDebug() << "Connecting to Deimos system";
267  cameras_connected = true;
268  }else {
269  ui->statusBar->showMessage("Couldn't find any cameras.");
270 
271  delete stereo_cam_deimos;
272  delete stereo_cam_phobos;
273  }
274 
275  if (cameras_connected) {
276  stereoCameraInit();
277  }

280 void MainWindow::stereoCameraInit() {
